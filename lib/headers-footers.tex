% use fancyhdr
\usepackage{fancyhdr}

% apply fancyhdr style on all pages except 'plain' specified ones (e.g. title
% page)
\pagestyle{fancy}

% clear default header & footer styles
\fancyhf{}

% and line rules as well
%\renewcommand{\headrulewidth}{0pt}
%\renewcommand{\footrulewidth}{0pt}

% make header bigger to remove warnings
%\setlength{\headheight}{15pt}

% add header/footer line rules
\renewcommand{\headrulewidth}{0.5pt}
\renewcommand{\footrulewidth}{0.5pt}

% remove numbering from sections in headers
\renewcommand{\sectionmark}[1]{\markboth{#1}{}}

% same for subsections
%\renewcommand{\subsectionmark}[1]{\markright{#1}}

% command shortcuts
\newcommand{\header}[1]{\textsf{\textit{\nouppercase{#1}}}}
\newcommand{\footer}[1]{#1}

% set header & footer
\lhead{\header{\leftmark}}
\rhead{\header{\rightmark}}

\cfoot{\footer{\thepage}}
\rfoot{\footer{Name}}
